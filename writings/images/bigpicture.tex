\usetikzlibrary{calc}
\begin{tikzpicture}
\tikzstyle{vecArrow} = [thick, decoration={markings,mark=at position
   1 with {\arrow[semithick]{open triangle 60}}},
   double distance=1.4pt, shorten >= 5.5pt,
   preaction = {decorate},
   postaction = {draw,line width=1.4pt, white,shorten >= 4.5pt}]
\tikzstyle{fancyArrow} = [<->,very thick,decorate,decoration={snake,amplitude=.4mm,segment length=2mm,post length=1.5mm, pre length=1.5mm}]

	\coordinate (top) at (0.0,2.3);
	\coordinate (bottom) at (0.0,-2.3);
	\coordinate (width) at (5.3,0);
	\coordinate (vspace) at (0,0.2);
	
	\coordinate (pos1) at ($(-5,0) + (bottom)$);
	\coordinate (pos2) at ($(0.7,0) + (bottom)$);
	\path (pos1) +($(width) + 2*(top)$) coordinate (rcorn1);
	\path (pos2) +($(width) + 2*(top)$) coordinate (rcorn2);
	
	\node (dummytop1) at ($(pos1)+2*(top)+0.5*(width)$) {};
	\node (dummytop2) at ($(pos2)+2*(top)+0.5*(width)$) {};
	\node (dummybott1) at ($(pos1)+0.5*(width)$) {};
	\node (dummybott2) at ($(pos2)+0.5*(width)$) {};

  \draw[thick,dashed] (pos1) rectangle (rcorn1);
  \draw[thick, dashed] (pos2) rectangle (rcorn2);

  \node (en) at ($(dummytop1.south)-(vspace)$)  {\textbf{ENTANGLEMENT}};
  \node (pr) at ($(dummytop2.south)-(vspace)$)  {\textbf{PRIVACY}};
  \node (dist) at ($(en.south) - (0,0.7)$)  {Entanglement distillation};
  \node (cka) at ($(pr.south) - (0,0.7)$)  {Classical key agreement};
  \node (be) at ($(dummybott1.north) + (vspace)$)  {Bound Entanglement};
  \node (bi) at ($(dummybott2.north) + (vspace)$)  {Bound Information};
  \draw[vecArrow] (dist) to (be);
  \draw[vecArrow] (cka) to (bi);
  \draw[fancyArrow] (dist) to (cka);
  \draw[fancyArrow] (be) to (bi);

\end{tikzpicture}
